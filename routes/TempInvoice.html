function generateSalesConfirmationEmailHTML(customerName, orderId, orderDate, items, totalAmount) { let itemsList = items.map(item => ` ${item.name} ${item.quantity} $${item.price} `).join(''); return `

Hi ${customerName},

Thank you for your order! Here is your sales confirmation:

Order ID: ${orderId}
Order Date: ${orderDate}

${itemsList}
Item Quantity Price

Total Amount: $${totalAmount}

We appreciate your business!

Best regards,
Your Company

`; } // 用法示例 const emailHTML = generateSalesConfirmationEmailHTML( "Alice", "12345", "2025-08-22", [ { name: "Product A", quantity: 2, price: 50 }, { name: "Product B", quantity: 1, price: 30 } ], 130 ); console.log(emailHTML);
